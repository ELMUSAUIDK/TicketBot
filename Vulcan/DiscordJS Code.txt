# -- Users --
message.member - Returns a guild user
message.author - Returns the author's discord name
message.author.id - Returns author's id
message.author.toString - Returns author's id
message.author.tag - Returns User#1234

# -- Guild --
message.guild.name - Get the guild name

# -- Messages --
message.content.includes("") - If the message contains a word
